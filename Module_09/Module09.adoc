def plotData(fileName):
    xVals, yvals = getData(fileName)
    xVals = pylab.array(xVals)
    yVals = pylab.array(yVals)
    xVals = xVals * 9.81    # acc. 중력으로 인함
    pylab.plot(xVals, yVals, 'bo', label = 'Measured displacements')
    labelPlot()
